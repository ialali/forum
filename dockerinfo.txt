# Set the base image
FROM golang:latest

# Set metadata labels
LABELS maintainers="names"
LABEL version="1.0.0"
LABEL description="Docker img for Go app"

# Set the working directory
WORKDIR /app

# Copy the application code
COPY . .

# Build the application
RUN go build -o main

# Expose the application port
EXPOSE 8005

# Set the command to run the app
CMD ["/app/main"]
